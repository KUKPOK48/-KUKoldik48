import tkinter as tk
          import random
          
          class GuessNumberGame:
              def __init__(self, root):
                  self.root = root
                  self.root.title("Игра 'Угадай число'")
                  
                  self.number_to_guess = random.randint(1, 100)
                  self.attempts = 0
          
                  self.label = tk.Label(root, text="Введите число от 1 до 100:", font=("Arial", 14))
                  self.label.pack(pady=10)
                  
                  self.entry = tk.Entry(root, font=("Arial", 14))
                  self.entry.pack(pady=5)
                  
                  self.submit_button = tk.Button(root, text="Угадать", font=("Arial", 14), command=self.check_guess)
                  self.submit_button.pack(pady=10)
                  
                  self.result_label = tk.Label(root, text="", font=("Arial", 14))
                  self.result_label.pack(pady=10)
                  
                  self.restart_button = tk.Button(root, text="Начать заново", font=("Arial", 14), command=self.restart_game)
                  self.restart_button.pack(pady=10)
                  self.restart_button.config(state=tk.DISABLED)
          
              def check_guess(self):
                  try:
                      guess = int(self.entry.get())
                      self.attempts += 1
                      if guess < self.number_to_guess:
                          self.result_label.config(text="Мое число больше!", fg="red")
                      elif guess > self.number_to_guess:
                          self.result_label.config(text="Мое число меньше!", fg="red")
                      else:
                          self.result_label.config(text=f"Поздравляю! Вы угадали число за {self.attempts} попыток!", fg="green")
                          self.submit_button.config(state=tk.DISABLED)
                          self.restart_button.config(state=tk.NORMAL)
                  except ValueError:
                      self.result_label.config(text="Пожалуйста, введите целое число.", fg="red")
          
              def restart_game(self):
                  self.number_to_guess = random.randint(1, 100)
                  self.attempts = 0
                  self.result_label.config(text="")
                  self.submit_button.config(state=tk.NORMAL)
                  self.restart_button.config(state=tk.DISABLED)
                  self.entry.delete(0, tk.END)
          
          root = tk.Tk()
          game = GuessNumberGame(root)
          root.mainloop()

                  cube.Position = cube.Position + Vector3.new(0, 5, 0)

                  cube.Position = cube.Position - Vector3.new(0, 5, 0)
              end
          end

              if cube.Transparency == 0 then
                  cube.Transparency = 0.5

                  cube.Transparency = 0
              end
